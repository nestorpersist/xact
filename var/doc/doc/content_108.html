<script> ::= {S? <cmds>}? S?
<cmds> ::= {{<simple cmd> S? ';' S?} | {<compound cmd> S?}}*
{{<simple cmd> {S? ';'}?} | <compound cmd>}
<simple cmd> ::= <short cmd> | <exp>
<compound cmd> ::= <long cmd> | <element>
<compound cmd> ::= {<id> ':'}? '/*' {text - {text '*/' text}} '*/'
<compound cmd> ::= {<id> ':'}? '//' {text - {text eol text}} eol
<short cmd> ::= '@' <id> S {<exps>}?
<long cmd> ::= <short cmd> S? '{' {S? <cmds>}? S? '}'
<element> ::= <start tag> S? {<xml> S?}* <end tag>
<element> ::= <empty tag>
<element> ::= '<?xml' {S <attr>}* S? '?>'
<element> ::= '<' {<short cmd> | <long cmd>} S? '/>'
<start tag> ::= '<' <extended id> {S <attr>}* S? '>'
<end tag> ::= '</' <extended id> S? '>'
<empty tag> ::= '<' <extended id> {S <attr>}* S? '/>'
<attr> ::= <extended id> S? '=' S? <string>
<xml> ::= <element> | <escape>
<xml> ::= text - {text {'<' | '&'} text}
<xml> ::= '<!--' {text - {text '-->' text}} '-->'
<escape> ::= '&' '#' {digit}+ ';'
<escape> ::= '&' '#' {'x' | 'X'} {digit | letter}+ ';'
<escape> ::= '&' <extended id> ';'
S ::= {' ' | tab | eol}+
<exps> ::= <exp> {S? ',' S? <exp>}*
<exp> ::= <qualified>
<exp> ::= <prefix op> S? {'#' S? <primary> S?}? <exp>
<exp> ::= <exp> S? <infix op> S? {'#' S? <primary> S?}? <exp>
<qualified> ::= <primary>
<qualified> ::= <qualified> S? '.' S? <name> {S? '#' S? <primary>}?
<qualified> ::= <qualified> S? '(' S? {<exps> S?}? ')' {S? '#' S? <primary>}?
<primary> ::= <name> | <int> | <float> | <string>
<primary> ::= '(' S? <exp> S? ')'
<primary> ::= '[' S? <exps> S? ']'
<prefix op> ::= '~' | '-' | '!' | '%' | '**'
<infix op> ::= '=' ':=' '..' '++' '~' '?'
<infix op> ::= '==' | '!=' | '<' | '>' | '<=' | '>='
<infix op> ::= '+' | '-' | '*' | '/' | '/+'
<infix op> ::= '&' | '|' | '^'
<name> ::= <id> {S? '[' S? {<exps> S?}? ']'}?
<id> ::= <basic id> | {'$' <extended id> '$'}
<basic id> ::= {letter | '_' | ':'}
{letter | '_' | ':' | digit}*
<extended id> ::= {letter | '_' | ':' | '.'}
{letter | '_' | ':' | '.' | '-' | digit}*
<string> ::= '"' {{schar - '"'} | <escape>}* '"'
<string> ::= ''' {{schar - '''} | <escape>}* '''
<int> ::= {digit}+
<float> ::= <int> '.' {<int>}? {'e' | 'E' {'+' | '-'}? <int>}?
char ::= letter | digit | '@' | '_' | '.' | '$' | '%' |
| '(' | ')' | '{' | '}' | '[' | ']' | ''' | '"' | '`' |
| '+' | '-' | '*' | '/' | '|' | '^' | '&' | '!' |
| ':' | '=' | '<' | '>' |
| '~' | '?' | '#' | ';' | ',' | '\'
schar ::= {char | ' '} - '&'
text ::= {char | S}*
eol ::= {CR}? LF
letter ::= 'a'..'z' | 'A'..'Z'
digit ::= '0'..'9'
